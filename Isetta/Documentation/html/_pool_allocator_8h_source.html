33 template <typename T>
34 PoolAllocator<T>::PoolAllocator(const SizeInt count) {
35  isErased = false;
36  capacity = count;
37  elementSize = sizeof(T);
38  memHead = MemoryAllocator::AllocateDefaultAligned(elementSize * capacity);
39  PtrInt address = reinterpret_cast<PtrInt>(memHead);
40  head = new (memHead) Node();
41  Node* cur = head;
42 
43  for (SizeInt i = 1; i < count; i++) {
44  address += elementSize;
45  Node* node = new (reinterpret_cast<void*>(address)) Node();
46  cur->next = node;
47  cur = cur->next;
48  }
49 }
